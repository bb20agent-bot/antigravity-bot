# VORA Ecosystem Nginx Configuration Template
# Path: /etc/nginx/sites-available/vora

server {
    listen 80;
    server_name voraswap.com vorascan.com voramini.com vorainvest.com www.voraswap.com www.vorascan.com www.voramini.com www.vorainvest.com;
    
    # Redirect all HTTP to HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    server_name voraswap.com vorascan.com voramini.com vorainvest.com www.voraswap.com www.vorascan.com www.voramini.com www.vorainvest.com;

    # SSL Configuration (To be handled by Certbot)
    # sudo certbot --nginx -d voraswap.com -d vorascan.com -d voramini.com -d vorainvest.com
    
    # Root directory for your Vite 'dist' folder
    root /var/www/vora/dist;
    index index.html;

    location / {
        # Support React/Vite routing
        try_files $uri $uri/ /index.html;
    }

    # API Proxy (if your backend server/vora_webhook_handler.py is on port 3001)
    location /api {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Logging
    access_log /var/log/nginx/vora_access.log;
    error_log /var/log/nginx/vora_error.log;
}
