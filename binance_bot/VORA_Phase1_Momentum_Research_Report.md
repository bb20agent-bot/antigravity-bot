### **알고리즘 상세**
```python
# Step 1: 각 타임프레임 수익률 계산
def calculate_returns(df, stock):
    df['1M Return'] = (close / close_1month_ago) - 1
    df['3M Return'] = (close / close_3month_ago) - 1
    df['6M Return'] = (close / close_6month_ago) - 1
    df['1Y Return'] = (close / close_1year_ago) - 1
    return df

# Step 2: Percentile 계산
def calculate_percentiles(df):
    for period in ['1M', '3M', '6M', '1Y']:
        df[f'{period} Percentile'] = df[f'{period} Return'].rank(pct=True)
    return df

# Step 3: HQM Score 계산
def calculate_hqm_score(df):
    df['HQM Score'] = df[['1M Percentile', '3M Percentile', 
                           '6M Percentile', '1Y Percentile']].mean(axis=1)
    return df

# Step 4: TOP 50 선택
def select_top_momentum(df, n=50):
    df_sorted = df.sort_values('HQM Score', ascending=False)
    top_50 = df_sorted.head(n)
    return top_50

# Step 5: Equal-Weight Portfolio
def calculate_position_size(portfolio_value, num_stocks=50):
    position_size = portfolio_value / num_stocks
    return position_size
```

pct_6m = ta.percentile_rank(returns_6m, 252)
pct_1y = ta.percentile_rank(returns_1y, 252)

hqm_score = (pct_1m + pct_3m + pct_6m + pct_1y) / 4

// HQM 필터 추가
hqm_strong = hqm_score > 70  // 상위 30%
hqm_weak = hqm_score < 30    // 하위 30%

# Step 1: 전일 대비 상승률 체크
def check_daily_momentum(stock, threshold=0.04):
    """
    전일 종가 대비 +4% 이상
    """
    prev_close = get_previous_close(stock)
    current_price = get_current_price(stock)
    gain = (current_price / prev_close) - 1
    
    return gain >= threshold

# Step 2: 15분 고점 돌파 확인
def check_breakout(stock):
    """
    9:30-9:45 사이 고점 돌파
    """
    high_15min = get_high_between('9:30', '9:45', stock)
    current_price = get_current_price(stock)
    
    return current_price > high_15min

# Step 3: 거래량 급증 확인
def check_volume_surge(stock, multiplier=2.0):
    """
    평균 거래량 대비 2배 이상
    """
    avg_volume_20d = get_avg_volume(stock, days=20)
    current_volume = get_current_volume(stock)
    
    return current_volume > avg_volume_20d * multiplier

# Step 4: 시가총액 필터
def check_market_cap(stock, min_cap=500_000_000):
    """
    최소 $500M (유동성 확보)
    """
    market_cap = get_market_cap(stock)
    return market_cap >= min_cap

# Step 5: 종합 진입 신호
def generate_buy_signal(stock):
    if not (9.75 <= current_hour <= 10.5):  // 9:45-10:30
        return False
    
    checks = [
        check_daily_momentum(stock, 0.04),
        check_breakout(stock),
        check_volume_surge(stock, 2.0),
        check_market_cap(stock, 500_000_000)
    ]
    
    return all(checks)

# Step 6: 청산 신호
def generate_sell_signal(stock, position):
    # 조건 1: 시간 청산 (3:45 PM)
    if current_time >= '15:45':
        return True
    
    # 조건 2: 손절 (-2%)
    entry_price = position.avg_entry_price
    current_price = get_current_price(stock)
    pnl_pct = (current_price / entry_price) - 1
    
    if pnl_pct <= -0.02:
        return True
    
    # 조건 3: 이익 실현 (+5%)
    if pnl_pct >= 0.05:
        return True
    
    return False
